import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import tensorflow as tf
from sklearn.preprocessing import LabelEncoder, OneHotEncoder
from sklearn.model_selection import train_test_split
from tensorflow.keras.callbacks import EarlyStopping
from tensorflow.keras.regularizers import l2

data = pd.read_excel(r"D:\L Leng\master of analytics\161777 Practical Data Mining\Group work\Group Project\Celebrity Data for PresentationProjects-20240416\training_celeb100x100.xlsx")
X = data.iloc[:, 1:].values.reshape(-1, 100, 100, 1)
y = data.iloc[:, 0].values

category_names = data.iloc[:, 0].unique()

encoder = LabelEncoder()
y = encoder.fit_transform(y)
y = y.reshape(-1, 1)

one_hot_encoder = OneHotEncoder()
y_onehot = one_hot_encoder.fit_transform(y).toarray()

X_train, X_val, y_train, y_val = train_test_split(X, y_onehot, stratify=y_onehot, test_size=0.2, random_state=42)

model = tf.keras.Sequential([
    tf.keras.layers.Conv2D(32, (3, 3), activation='relu', input_shape=(100, 100, 1), kernel_regularizer=l2(0.001)),
    tf.keras.layers.MaxPooling2D((2, 2)),
    tf.keras.layers.Dropout(0.25),
    tf.keras.layers.Conv2D(64, (3, 3), activation='relu', kernel_regularizer=l2(0.0002)),
    tf.keras.layers.MaxPooling2D((2, 2)),
    tf.keras.layers.Dropout(0.25),
    tf.keras.layers.Flatten(),
    tf.keras.layers.Dense(256, activation='relu', kernel_regularizer=l2(0.0002)),
    tf.keras.layers.Dropout(0.5),
    tf.keras.layers.Dense(y_onehot.shape[1], activation='softmax')
])

model.compile(optimizer=tf.keras.optimizers.Adam(learning_rate=0.0002),
              loss='categorical_crossentropy',
              metrics=['accuracy'])

early_stop = EarlyStopping(monitor='val_loss', patience=10, restore_best_weights=True)

history = model.fit(X_train, y_train,
                    epochs=500,
                    validation_data=(X_val, y_val),
                    callbacks=[early_stop],
                    verbose=2)

train_loss = history.history['loss']
val_loss = history.history['val_loss']
epochs = range(1, len(train_loss) + 1)
plt.plot(epochs, train_loss, 'b', label='Training loss')
plt.plot(epochs, val_loss, 'r', label='Validation loss')
plt.title('Training and validation loss')
plt.xlabel('Epochs')
plt.ylabel('Loss')
plt.legend()
plt.show()

test_data = pd.read_csv(r"D:\L Leng\master of analytics\161777 Practical Data Mining\Group work\Group Project\Celebrity Data for PresentationProjects-20240416\unlabelledtest_celeb100x100.csv")
X_test = test_data.iloc[:, :].values.reshape(-1, 100, 100, 1)

y_pred = model.predict(X_test)
print(y_pred)

y_pred_classes = np.argmax(y_pred, axis=1)

category_names = data.iloc[:, 0].unique()
